<script>{`
  setTimeout(() => {
    const nameField = document.getElementById('name-field');
    if (nameField) {
      nameField.appendChild(window.createInput({
        label: 'Nome Completo',
        placeholder: 'Digite seu nome',
        required: true,
        id: 'val-name'
      }));
    }

    const emailField = document.getElementById('email-field');
    if (emailField) {
      emailField.appendChild(window.createInput({
        label: 'E-mail Institucional',
        placeholder: 'email@orgao.gov.br',
        type: 'email',
        id: 'val-email'
      }));
    }

    const cpfField = document.getElementById('cpf-field');
    if (cpfField) {
      cpfField.appendChild(window.createInput({
        label: 'CPF',
        placeholder: '000.000.000-00',
        id: 'val-cpf'
      }));
    }

    const categoryField = document.getElementById('category-field');
    if (categoryField) {
      categoryField.appendChild(window.createSelect({
        label: 'Categoria do Processo',
        options: [
          { label: 'Administrativo', value: 'adm' },
          { label: 'Financeiro', value: 'fin' },
          { label: 'Recursos Humanos', value: 'rh' }
        ],
        id: 'val-category'
      }));
    }

    const descriptionField = document.getElementById('description-field');
    if (descriptionField) {
      descriptionField.appendChild(window.createTextarea({
        label: 'Descrição Detalhada',
        placeholder: 'Descreva o motivo da abertura do processo...',
        maxlength: 500,
        showCounter: true,
        id: 'val-desc'
      }));
    }

    const typeField = document.getElementById('type-field');
    if (typeField) {
      typeField.appendChild(window.createRadioGroup({
        label: 'Prioridade',
        name: 'prioridade-val',
        options: [
          { label: 'Normal', value: 'normal' },
          { label: 'Urgente', value: 'urgente' }
        ],
        selectedValue: 'normal',
        orientation: 'horizontal'
      }));
    }

    const notificationField = document.getElementById('notification-field');
    if (notificationField) {
      notificationField.appendChild(window.createCheckboxGroup({
        label: 'Notificações',
        options: [
          { label: 'E-mail', value: 'email' },
          { label: 'SMS', value: 'sms' }
        ],
        orientation: 'horizontal'
      }));
    }

    const cancelBtn = document.getElementById('cancel-btn');
    if (cancelBtn) {
      cancelBtn.appendChild(window.createButton({
        label: 'Cancelar',
        variant: 'ghost'
      }));
    }

    const saveBtn = document.getElementById('save-btn');
    if (saveBtn) {
      saveBtn.appendChild(window.createButton({
        label: 'Salvar Requerimento',
        variant: 'primary'
      }));
    }

    // Resumo
    const pair1 = document.getElementById('pair-1');
    if (pair1) pair1.appendChild(window.createCardDataPair('Status do Sistema', 'Operacional'));
    
    const pair2 = document.getElementById('pair-2');
    if (pair2) pair2.appendChild(window.createCardDataPair('Data Prevista', '14/02/2026'));

    const pair3 = document.getElementById('pair-3');
    if (pair3) pair3.appendChild(window.createCardDataPair('Responsável', 'Departamento de Protocolo'));

  }, 100);
`}</script>
